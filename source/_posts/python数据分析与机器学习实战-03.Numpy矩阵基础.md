```
#矩阵乘法
a = np.array((
    [1,5],
    [0,1]))
b = np.array((
    [2,0],
    [3,4]))
print(a)
print(b)
print("-------")
#对应位置相乘
print(a*b)
[[1 5]
 [0 1]]
[[2 0]
 [3 4]]
-------
[[2 0]
 [0 4]]
```
