import pandas as pd
import matplotlib.pyplot as plt

data = pd.read_excel("2017.12.01-2018.12.01日销售汇总.xls")
sales = data[365:501]["实收金额"]
date = data[365:501]["日期"]
print(date)
plt.plot(date,sales)
plt.xticks(rotation=45)
plt.show()
